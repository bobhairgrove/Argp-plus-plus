1112  , char** vargv
1113  , ArgppBase* parent = NULL)
1114  {
1115  ArgppBase* argpp_root = ArgppBase::getRootInstance();
1116  T* retval = NULL;
1117  //----------------------------------------------------
1118  // If parent == NULL, the root instance becomes the
1119  // parent.
1120  //
1121  // If argpp_root == NULL, the newly created object
1122  // becomes the root instance.
1123  //----------------------------------------------------
1124  bool parent_is_valid = true;
1125 
1126  if (!parent) {
1127  parent = argpp_root;
1128  } else {
1129  parent_is_valid = ArgppBase::isParentValid(parent);
1130  }
1131 
1132  if (parent_is_valid) {
1133  bool ok = (argpp_root && (ArgppBase::getArgc() == cargc)
1134  && (ArgppBase::getArgv() == vargv))
1135  || !argpp_root;
1136  //------------------------------------------------------
1137  // We don't catch any exceptions generated by user code;
1138  // instead, we let them propagate up to the caller.
1139  //------------------------------------------------------
1140  if (ok) {
1141  T* pObj = new T( cargc, vargv, parent );
1142  if (pObj && !argpp_root) {
1143  ArgppBase::installArgppRootInstance(pObj);
1144  }
1145  retval = pObj;
1146  }
1147  }
1148  return retval;
1149  }
1150  //-----------------------------------------------------------------------
1151  // Overloaded version so that clients don't have to pass the command line
1152  // arguments each time they create an object. Once the root parser has
1153  // been created, the argc and argv parameters are stored as static data
1154  // and passed on to all of the child parsers.
1155  //-----------------------------------------------------------------------
1165  static T* createParser(ArgppBase * parent = NULL)
1166  {
1167  ArgppBase* root = ArgppBase::getRootInstance();
1168  if (root) {
1169  return createParser(ArgppBase::getArgc()
1170  , ArgppBase::getArgv()
1171  , parent);
1172  }
1173  return NULL;
1174  }
