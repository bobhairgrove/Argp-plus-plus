120 error_t ArgppBaseImpl::parserRouter(int key, char *arg, argp_state *state) {
121 
122  static error_t retval = ARGPP_UNKNOWN;
123  //-----------------------------------------------------------------
124  // Put everything in a separate scope so that locals are cleaned up
125  // if we have to call exit():
126  //-----------------------------------------------------------------
127  {
128  //-------------------------------------------------------------------------
129  // The argp_parse() function passes on the "void* input" argument in the
130  // state->input member. Our function casts the state->input member to an
131  // ArgppBase* since we passed it the "this" pointer in the initial call
132  // to argp_parse() as the "void* input" argument.
133  //-------------------------------------------------------------------------
134  if (state && state->input) {
135 
136  ArgppBase *pThis = static_cast<ArgppBase*>(state->input);
137  ArgppBaseImpl *p = pThis->pimpl_;
138 
139  switch (key) {
140  case ARGP_KEY_INIT:
141  //-------------------------------------------------------------------------
142  // Argp code always calls the parser function with the key ARGP_KEY_INIT
143  // as the very first call to the callback routine.
144  //
145  // What needs to be done first here is to set up the child_inputs[] vector
146  // for each immediate child of the object whose "this" pointer is passed in
147  // state->input. Afterwards, the argp code calls the parserRouter for
148  // each child in succession with the corresponding child_input value
149  // copied into state->input. If the children have any children of their
150  // own, they do the same by setting up their child_inputs when parserRouter
151  // is called with key == ARGP_KEY_INIT.
152  //
153  // We depend on the sorting order of std::map, which uses the address of
154  // the POD argp_child vector's element as the key, as well as the C++
155  // standard's guarantee that elements of a vector are stored in contiguous
156  // memory. The map's value element (member "second") contains the "this"
157  // pointers to the corresponding children.
158  //-------------------------------------------------------------------------
159  {
160  size_t len = p->map_of_children_.size();
161  MapOfChildren::const_iterator cit = p->map_of_children_.begin();
162  for(size_t i=0; i<len; ++i, ++cit)
163  {
164  state->child_inputs[i] = static_cast<void*>(cit->second);
165  }
166  }
167  //-----------------------------------------
168  // Let the derived class do some additional
169  // initialization work here, if any:
170  //-----------------------------------------
171  retval = pThis->initImpl(state);
172  break;
173  case ARGP_KEY_ERROR:
174  retval = pThis->errorImpl(state);
175  break;
176  case ARGP_KEY_FINI:
177  retval = pThis->finiImpl(state);
178  break;
179  case ARGP_KEY_ARG:
180  retval = pThis->keyArgImpl(arg, state);
181  break;
182  case ARGP_KEY_ARGS:
183  retval = pThis->keyArgsImpl(state);
184  break;
185  case ARGP_KEY_NO_ARGS:
186  retval = pThis->keyNoArgsImpl(state);
187  break;
188  case ARGP_KEY_SUCCESS:
189  retval = pThis->keySuccessImpl(state);
190  break;
191  case ARGP_KEY_END:
192  retval = pThis->keyEndImpl(state);
193  break;
194  default:
195  retval = pThis->parserImpl(key, arg, state);
196  break;
197  }
198  }
199  }
200  //------------------------------------------------------
201  // This is where we can safely call exit() if necessary:
202  //------------------------------------------------------
203  if (argpp_should_exit_) {
204  clearAllParsers();
205  exit(error_code_from_argp_);
206  }
207  return retval;
208 }

210 //--------------------------------------------------------------------
211 char *ArgppBaseImpl::helpFilter(int key, const char *text, void *input)
212 {
213  //-------------------------------------------------------
214  // If the derived class does not return a different string,
215  // we have to return the string passed in the "text"
216  // argument, but first cast away its const qualifier:
217  //-------------------------------------------------------
218  char *retval = const_cast<char*>(text);
219 
220  //-------------------------------------------------------
221  // These are used in the code which calls malloc()
222  // later on (that is, if the derived implementation of
223  // helpFilterImpl() returns a string we have to copy):
224  //-------------------------------------------------------
225  static const size_t DEFAULT_MEM_BLK_SIZE = 8;
226  static const size_t MAX_BLK_SIZE = ArgppBaseImpl::help_max_blk_size_;
227 
228  if (input) {
229  ArgppBase * pThis = static_cast<ArgppBase*>(input);
230  std::string newtxt = pThis->helpFilterImpl(key, text);
231 
232  if (!newtxt.empty() && (!text || newtxt.compare(text) != 0)) {
233  //-------------------------------
234  // argp wants a malloc'ed string:
235  //-------------------------------
236  size_t memsize = DEFAULT_MEM_BLK_SIZE;
237  size_t len = newtxt.size();
238 
239  if (len < MAX_BLK_SIZE) {
240  // This allocates slightly more memory than
241  // the actual required length plus the
242  // terminating '\0' char:
243  ++len;
244  while (memsize < len) {
245  memsize += DEFAULT_MEM_BLK_SIZE;
246  }
247  --len;
248  void *pmem = malloc(memsize);
249  if (pmem) {
250  char *pTarget = static_cast<char*>(pmem);
251  const char *pBegin = newtxt.c_str();
252  const char *pEnd = pBegin + len;
253  std::copy(pBegin, pEnd, pTarget);
254  pTarget[len] = '\0';
255  retval = pTarget;
256  }
257  }
258  }
259  }
260  return retval;
261 }

386 //-----------------------------------------------------------------------------
387 void ArgppBaseImpl::argppError(const argp_state *state, const std::string &errmsg)
388 {
389  FILE* stream = NULL;
390 
391  if (state) {
392  unsigned int f = state->flags;
393  std::string prg_name;
394  std::string msg;
395 
396  if (state->err_stream) {
397  stream = state->err_stream;
398  } else {
399  stream = stderr;
400  }
401 
402  if (state->name) {
403  prg_name = state->name;
404  } else if (state->argv && state->argv[0]) {
405  std::string n = state->argv[0];
406  size_t pos = n.rfind(ARGPP_DIR_SEPARATOR);
407  if (pos != std::string::npos) {
408  prg_name = n.substr(pos+1);
409  } else {
410  prg_name = n;
411  }
412  }
413  //--------------------------------------
414  // strip off the './' due to program
415  // invocation from the shell, if any:
416  //--------------------------------------
417  if (prg_name.substr(0,2) == "./") {
418  prg_name = prg_name.substr(2);
419  }
420  msg = prg_name + ": " + errmsg + "\n";
421  fputs(msg.c_str(), stream);
422 
423  //--------------------------------------------------------
424  // Prevent argp_state_help() from calling exit() so as not
425  // to leak memory in the errmsg argument to this function:
426  //--------------------------------------------------------
427  struct argp_state s = { state->root_argp
428  , state->argc
429  , state->argv
430  , state->next
431  , state->flags | ARGP_NO_EXIT
432  , state->arg_num
433  , state->quoted
434  , state->input
435  , state->child_inputs
436  , state->hook
437  , state->name
438  , state->err_stream
439  , state->out_stream
440  , state->pstate};
441 
442  argp_state_help (&s, stream, ARGP_HELP_STD_ERR);
443 
444  error_code_from_argp_ = 1;
445 
446  if ((f & ARGP_NO_EXIT) == 0) {
447  argpp_should_exit_ = true;
448  }
449  }
450 }

975 //-----------------------------------------------------------------------------
976 void ArgppBaseImpl::setupChildren()
977 {
978  //---------------------------------------------------
979  // Each child should set up its options,
980  // its own children and its main_argp_
981  // member in that order.
982  //
983  // The parent sets up its vectors only after
984  // the children are done with their work.
985  //---------------------------------------------------
986  size_t len_children = children_.size();
987 
988  //---------------------------------------------------
989  // I know, it's paranoid, but...
990  //---------------------------------------------------
991  children_vec_.clear();
992 
993  for (size_t i=0; i<len_children; ++i) {
994 
995  children_[i]->pimpl_->prepareParserVecs();
996 
997  //-----------------------------------------------
998  // Set up the vector of argp_child structs now:
999  //-----------------------------------------------
1000  children_vec_.push_back(children_[i]->pimpl_->child_argp_);
1001 
1002  //-----------------------------------------------
1003  // Map the "this" pointers so that parserRouter()
1004  // can find the right objects later:
1005  //-----------------------------------------------
1006  argp_child* ac = &children_vec_[i];
1007  ArgppBase* pc = children_[i];
1008 
1009  map_of_children_.insert(std::make_pair(ac, pc));
1010  }
1011 
1012  //---------------------------------------------------------
1013  // Make sure that the argp_child vector is NULL-terminated:
1014  //---------------------------------------------------------
1015  if (!children_vec_.empty()) {
1016  children_vec_.push_back(NULL_ARGP_CHILD);
1017  }
1018 
1019  //------------------------------------
1020  // Skip this step for the root parser:
1021  //------------------------------------
1022  if (holders_parent_) {
1023  child_argp_.argp = &main_argp_;
1024  //-----------------------------------------------
1025  // Use the flags from the parent if ours are 0,
1026  // unless root is our parent in which case we use
1027  // the static member argp_flags_:
1028  //-----------------------------------------------
1029  if (!child_argp_.flags) {
1030  if (!holders_parent_->pimpl_->holders_parent_) {
1031  child_argp_.flags = static_cast<int>(argp_flags_);
1032  } else {
1033  child_argp_.flags = holders_parent_->childFlags();
1034  }
1035  }
1036  child_argp_.header = child_header_.empty()
1037  ? (child_header_is_empty_cstring_ ? NULL_CSTRING : 0)
1038  : child_header_.c_str();
1039  // child_argp_.group ... is already set; if not, it is == 0 ...
1040  }
1041 
1042  // do the main_argp_ now:
1043  main_argp_.options = option_vec_.empty() ? NULL : &option_vec_[0];
1044  main_argp_.parser = parserRouter;
1045  main_argp_.args_doc = usage_msg_.empty() ? NULL : usage_msg_.c_str();
1046  main_argp_.doc = doc_msg_.empty() ? NULL : doc_msg_.c_str();
1047  main_argp_.children = children_vec_.empty() ? NULL : &children_vec_[0];
1048  main_argp_.help_filter = installed_help_filter_ ? helpFilter : NULL;
1049  main_argp_.argp_domain = argp_domain_.empty() ? NULL : argp_domain_.c_str();
1050 }
